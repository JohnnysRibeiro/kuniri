def write_log(pMessage)
  File.open(@log_path, 'a') do |file|
    file.write("="*10 + "\n")
    file.write("time: " + Time.now.strftime("%d/%m/%Y %H:%M:%S") + "\n")
    file.write("message: " + pMessage + "\n")
  end
end
