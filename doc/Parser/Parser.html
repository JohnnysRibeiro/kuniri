def start_parser
  # TODO: handle the case of filePAth is empty
  @log.write_log("Debug: START FIRST PARSER")
  @filesPath.each do |file|
    language = @factory.get_language(@language)
    language.analyse_source(file)
    @fileLanguage.push(language)
  end
end
